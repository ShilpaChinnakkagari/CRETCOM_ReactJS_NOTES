function App() {

  let a = 15;
  let b = 5;

  return (
    <div>

      <h1>React JS Expressions</h1>

      <h2>Arithmetic Operations</h2>

      <p>Addition : {a + b}</p>
      <p>Subtraction : {a - b}</p>
      <p>Multiplication : {a * b}</p>
      <p>Division : {a / b}</p>
      <p>Modulus : {a % b}</p>

      <h2>String Expression</h2>

      <p>
        {"Hello " + "React JS"}
      </p>


      <h2>Variable Expression</h2>

      <p>
        Value of A : {a}
      </p>

      <p>
        Value of B : {b}
      </p>


      <h2>Conditional Expression</h2>

      <p>
        {a > b ? "A is Greater" : "B is Greater"}
      </p>


      <h2>Array Expression</h2>

      <p>
        {["HTML", "CSS", "React"][2]}
      </p>


      <h2>Object Expression</h2>

      <p>
        Student Name : {({name:"Shilpa", age:21}).name}
      </p>

    </div>
  );
}

export default App;

Getters and Setters in React
=================================
import { useState } from "react";

function Student(){

    // state variable (getter)
    const [name, setName] = useState("Shilpa");


    // setter function
    function changeName(){
        setName("Anu");
    }


    return(
        <div>

            <h1>Getter and Setter in Function Component</h1>

            {/* Getter */}
            <h2>Name : {name}</h2>


            {/* Setter */}
            <button onClick={changeName}>
                Change Name
            </button>

        </div>
    );
}

export default Student;
