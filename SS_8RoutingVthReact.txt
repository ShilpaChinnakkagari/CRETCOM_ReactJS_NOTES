Method - 2 (USing useEffect)
-----------
keyBoardEventConsole.jsx
------------------------
import { useState, useEffect } from "react";

function MultipleDependency() {

    const [text, setText] = useState("");
    const [keyUp, setKeyUp] = useState("");
    const [keyDown, setKeyDown] = useState("");

    useEffect(() => {

        console.log("Text Changed :", text);
        console.log("Key Up :", keyUp);
        console.log("Key Down :", keyDown);

    }, [text, keyUp, keyDown]);


    return (
        <div>

            <h2>useEffect Multiple Dependencies</h2>

            <input
                type="text"
                placeholder="Enter something"
                onChange={(e) => setText(e.target.value)}
                onKeyUp={(e) => setKeyUp(e.key)}
                onKeyDown={(e) => setKeyDown(e.key)}
            />

        </div>
    );
}

export {MultipleDependency};
